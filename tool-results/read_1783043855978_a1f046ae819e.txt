   174тЖТ   174тЖТfunction formatRelativeTime(dateStr: string) {
   175тЖТ   175тЖТ  const date = new Date(dateStr)
   176тЖТ   176тЖТ  const diff = Date.now() - date.getTime()
   177тЖТ   177тЖТ  const seconds = Math.floor(diff / 1000)
   178тЖТ   178тЖТ  const minutes = Math.floor(seconds / 60)
   179тЖТ   179тЖТ  const hours = Math.floor(minutes / 60)
   180тЖТ   180тЖТ  const days = Math.floor(hours / 24)
   181тЖТ   181тЖТ  if (days > 0) return `${days}d ago`
   182тЖТ   182тЖТ  if (hours > 0) return `${hours}h ago`
   183тЖТ   183тЖТ  if (minutes > 0) return `${minutes}m ago`
   184тЖТ   184тЖТ  return 'just now'
   185тЖТ   185тЖТ}
